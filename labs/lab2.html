    def plot_pb_side_by_side(img):
        grey = cv.cvtColor(img, cv.COLOR_RGB2GRAY)
        grey_3_channel = cv.cvtColor(grey, cv.COLOR_GRAY2BGR)
        horizontal_stack = np.hstack((img, grey_3_channel))
        gcv_imshow(horizontal_stack)
